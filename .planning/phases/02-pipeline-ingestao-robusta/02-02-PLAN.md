type OnboardingStatus =
  | 'pending'        // recebido pelo webhook, aguardando processamento
  | 'in_progress'    // cliente ainda preenchendo (admin pode receber updates parciais futuramente)
  | 'completed'      // cliente terminou onboarding mas processor/Jarvis ainda não rodou

  | 'live'           // tenant criado e funcionando
  | 'failed'         // falha terminal
  | 'needs_review';  // requer intervenção manual

    2. **Helper** `api/_lib/outbox.ts`:

       ```ts
       import { getServiceSupabase } from './supabase';
       import type { OnboardingPayload } from 'pipeelo-onboarding-contracts';

       export interface OutboxRow {
         id: string;
         session_id: string;
         target_url: string;
         payload: OnboardingPayload;
         status: 'pending' | 'in_flight' | 'delivered' | 'failed';
         attempt_count: number;
         max_attempts: number;
         last_error: string | null;
         next_retry_at: string;
         delivered_at: string | null;
       }

       /**
        * Idempotent insert — se já existe row para session_id, retorna existente.
        * Use ANTES de tentar deliver.
        */
       export async function enqueueOutbox(args: {
         sessionId: string;
         targetUrl: string;
         payload: OnboardingPayload;
       }): Promise<OutboxRow> {
         const sb = getServiceSupabase();
         // upsert: cria pending se novo, mantém row existente se já houver
         const { data, error } = await sb
           .from('webhook_outbox')
           .upsert(
             {
               session_id: args.sessionId,
               target_url: args.targetUrl,
               payload: args.payload,
               status: 'pending',
               next_retry_at: new Date().toISOString(),
             },
             { onConflict: 'session_id', ignoreDuplicates: false }
           )
           .select()
           .single();
         if (error) throw new Error(`enqueueOutbox failed: ${error.message}`);
         return data as OutboxRow;
       }

       export async function markInFlight(id: string): Promise<void> {
         const sb = getServiceSupabase();
         await sb.from('webhook_outbox')
           .update({ status: 'in_flight' })
           .eq('id', id)
           .eq('status', 'pending'); // optimistic — não sobrescreve in_flight de outro worker
       }

       export async function markDelivered(id: string): Promise<void> {
         const sb = getServiceSupabase();
         await sb.from('webhook_outbox').update({
           status: 'delivered',
           delivered_at: new Date().toISOString(),
         }).eq('id', id);
       }

       /**
        * Marca como pending de novo com next_retry_at calculado por backoff exponencial.
        * Se attempt_count+1 >= max_attempts → marca failed.
        */
       export async function markFailedAttempt(id: string, error: string, currentAttempts: number, maxAttempts: number): Promise<void> {
         const sb = getServiceSupabase();
         const nextAttempt = currentAttempts + 1;
         if (nextAttempt >= maxAttempts) {
           await sb.from('webhook_outbox').update({
             status: 'failed',
             attempt_count: nextAttempt,
             last_error: error.slice(0, 500),
           }).eq('id', id);
           return;
         }
         // Backoff exponencial: 30s * 2^attempt + jitter (até 8h)
         const baseMs = 30_000 * Math.pow(2, nextAttempt);
         const jitter = Math.random() * 0.3 * baseMs;
         const delayMs = Math.min(baseMs + jitter, 8 * 60 * 60 * 1000);
         const nextRetryAt = new Date(Date.now() + delayMs).toISOString();
         await sb.from('webhook_outbox').update({
           status: 'pending',
           attempt_count: nextAttempt,
           last_error: error.slice(0, 500),
           next_retry_at: nextRetryAt,
         }).eq('id', id);
       }

       /** Tenta entregar payload via fetch. Retorna { ok, status, body }. */
       export async function deliverOutbox(row: OutboxRow): Promise<{ ok: boolean; status?: number; body?: string }> {
         try {
           const resp = await fetch(row.target_url, {
             method: 'POST',
             headers: {
               'Content-Type': 'application/json',
               'Authorization': `Bearer ${process.env.ONBOARDING_WEBHOOK_TOKEN ?? ''}`,
               'Idempotency-Key': row.session_id,
             },
             body: JSON.stringify(row.payload),
             keepalive: true,
             signal: AbortSignal.timeout(15_000),
           });
           if (!resp.ok) {
             const body = await resp.text().catch(() => '');
             return { ok: false, status: resp.status, body: body.slice(0, 500) };
           }
           return { ok: true, status: resp.status };
         } catch (err: any) {
           return { ok: false, body: String(err?.message ?? err).slice(0, 500) };
         }
       }
       ```

       ```ts
       import type { VercelRequest, VercelResponse } from '@vercel/node';
       import { getServiceSupabase } from '../_lib/supabase';
       import { deliverOutbox, markDelivered, markFailedAttempt, markInFlight, type OutboxRow } from '../_lib/outbox';

       const BATCH_SIZE = 25;

       export default async function handler(req: VercelRequest, res: VercelResponse) {
         // Auth
         if (req.headers.authorization !== `Bearer ${process.env.CRON_SECRET ?? ''}`) {
           return res.status(401).json({ error: 'unauthorized' });
         }

         const sb = getServiceSupabase();
         const { data: pending, error } = await sb
           .from('webhook_outbox')
           .select('*')
           .eq('status', 'pending')
           .lte('next_retry_at', new Date().toISOString())
           .order('next_retry_at', { ascending: true })
           .limit(BATCH_SIZE);

         if (error) {
           return res.status(500).json({ error: 'db_error', message: error.message });
         }

         const results = await Promise.allSettled(
           (pending as OutboxRow[]).map(async (row) => {
             await markInFlight(row.id);
             const r = await deliverOutbox(row);
             if (r.ok) {
               await markDelivered(row.id);
               return { id: row.id, status: 'delivered' };
             }
             await markFailedAttempt(row.id, `${r.status ?? 'net'}: ${r.body ?? ''}`, row.attempt_count, row.max_attempts);
             return { id: row.id, status: 'retry_scheduled' };
           })
         );

         const summary = results.reduce(
           (acc, r) => {
             if (r.status === 'fulfilled') {
               acc[r.value.status] = (acc[r.value.status] ?? 0) + 1;
             } else {
               acc.errored = (acc.errored ?? 0) + 1;
             }
             return acc;
           },
           {} as Record<string, number>
         );

         return res.status(200).json({
           ok: true,
           processed: pending?.length ?? 0,
           summary,
         });
       }
       ```

       ```ts
       export type OnboardingStatus =
         | 'pending'
         | 'in_progress'
         | 'completed'
         | 'processing'
         | 'live'
         | 'failed'
         | 'needs_review';

       export const ALL_STATUSES: OnboardingStatus[] = [
         'pending', 'in_progress', 'completed', 'processing', 'live', 'failed', 'needs_review',
       ];

       const TRANSITIONS: Record<OnboardingStatus, OnboardingStatus[]> = {
         pending: ['in_progress', 'completed', 'failed'],
         in_progress: ['completed', 'failed', 'pending'],
         completed: ['processing', 'failed'],
         processing: ['live', 'needs_review', 'failed'],
         needs_review: ['processing', 'failed', 'pending'],
         failed: ['pending'],   // só ressuscitação manual
         live: [],              // terminal
       };

       export function canTransition(from: OnboardingStatus, to: OnboardingStatus): boolean {
         return TRANSITIONS[from]?.includes(to) ?? false;
       }

       export function assertTransition(from: OnboardingStatus, to: OnboardingStatus): void {
         if (!canTransition(from, to)) {
           throw new InvalidTransitionError(from, to);
         }
       }

       export class InvalidTransitionError extends Error {
         constructor(public from: OnboardingStatus, public to: OnboardingStatus) {
           super(`Invalid transition: ${from} → ${to}`);
           this.name = 'InvalidTransitionError';
         }
       }

       export function isTerminal(status: OnboardingStatus): boolean {
         return TRANSITIONS[status].length === 0;
       }
       ```
