    import { HttpError, type HttpRequest } from './types';

    const DEFAULT_TIMEOUT = 15000;
    const DEFAULT_RETRIES = 2;
    const RETRY_BASE_MS = 200;

    export type HttpResponse<T> = {
      ok: true;
      status: number;
      data: T;
      headers: Headers;
    };

    export async function callExternal<T = unknown>(req: HttpRequest): Promise<HttpResponse<T>> {
      const retries = req.retries ?? DEFAULT_RETRIES;
      const timeoutMs = req.timeoutMs ?? DEFAULT_TIMEOUT;
      let lastErr: HttpError | null = null;

      for (let attempt = 0; attempt <= retries; attempt++) {
        const controller = new AbortController();
        const timer = setTimeout(() => controller.abort(), timeoutMs);
        try {
          const res = await fetch(req.url, {
            method: req.method,
            headers: { 'content-type': 'application/json', ...(req.headers ?? {}) },
            body: req.body !== undefined ? JSON.stringify(req.body) : undefined,
            signal: controller.signal,
          });
          clearTimeout(timer);

          if (res.status >= 200 && res.status < 300) {
            const text = await res.text();
            const data = text ? JSON.parse(text) : null;
            return { ok: true, status: res.status, data: data as T, headers: res.headers };
          }

          // 4xx => não retenta (exceto 408/429 que são retriable)
          const retriable = res.status >= 500 || res.status === 408 || res.status === 429;
          const bodyText = await res.text().catch(() => '');
          lastErr = new HttpError(
            res.status >= 500 ? 'http_5xx' : 'http_4xx',
            res.status,
            retriable,
            `HTTP ${res.status} ${req.method} ${req.url}`,
            bodyText,
          );
          if (!retriable) throw lastErr;
        } catch (err) {
          clearTimeout(timer);
          if (err instanceof HttpError) {
            if (!err.retriable) throw err;
            lastErr = err;
          } else if ((err as Error).name === 'AbortError') {
            lastErr = new HttpError('timeout', null, true, `Timeout after ${timeoutMs}ms ${req.method} ${req.url}`);
          } else {
            lastErr = new HttpError('network', null, true, (err as Error).message);
          }
        }
        if (attempt < retries) {
          const delay = RETRY_BASE_MS * Math.pow(2, attempt) + Math.random() * 100;
          await new Promise(r => setTimeout(r, delay));
        }
      }
      throw lastErr ?? new HttpError('network', null, true, 'unreachable');
    }
    ```

    import { createHash } from 'node:crypto';
    import { getJarvisSupabase } from './supabase';

    /** JSON canonicalization: ordena keys de objects recursivamente; preserva ordem de arrays. */
    export function canonicalJson(value: unknown): string {
      const seen = new WeakSet();
      const stringify = (v: unknown): string => {
        if (v === null) return 'null';
        if (typeof v === 'number' || typeof v === 'boolean') return JSON.stringify(v);
        if (typeof v === 'string') return JSON.stringify(v);
        if (Array.isArray(v)) return '[' + v.map(stringify).join(',') + ']';
        if (typeof v === 'object') {
          if (seen.has(v as object)) throw new Error('canonicalJson: circular reference');
          seen.add(v as object);
          const keys = Object.keys(v as object).sort();
          return '{' + keys.map(k => JSON.stringify(k) + ':' + stringify((v as Record<string, unknown>)[k])).join(',') + '}';
        }
        return JSON.stringify(v ?? null);
      };
      return stringify(value);
    }

    export function hashArgs(args: unknown): string {
      return createHash('sha256').update(canonicalJson(args)).digest('hex');
    }

    export type IdempotencyResult<T> = { data: T; idempotentHit: boolean };

    export async function withIdempotency<T>(
      ctx: { sessionId: string; tool: string },
      args: unknown,
      fn: () => Promise<T>,
    ): Promise<IdempotencyResult<T>> {
      const supabase = getJarvisSupabase();
      const args_hash = hashArgs(args);

      // 1) Lookup cache
      const { data: cached, error: lookupErr } = await supabase
        .from('idempotency_keys')
        .select('result')
        .eq('session_id', ctx.sessionId)
        .eq('tool', ctx.tool)
        .eq('args_hash', args_hash)
        .maybeSingle();
      if (lookupErr) throw new Error(`idempotency lookup failed: ${lookupErr.message}`);
      if (cached) return { data: cached.result as T, idempotentHit: true };

      // 2) Execute fn (errors propagate, no cache write)
      const data = await fn();

      // 3) Persist cache (upsert para tolerar race condition entre 2 workers concorrentes)
      const { error: upsertErr } = await supabase
        .from('idempotency_keys')
        .upsert(
          { session_id: ctx.sessionId, tool: ctx.tool, args_hash, result: data as object },
          { onConflict: 'session_id,tool,args_hash', ignoreDuplicates: true },
        );
      if (upsertErr) throw new Error(`idempotency upsert failed: ${upsertErr.message}`);

      return { data, idempotentHit: false };
    }
    ```
