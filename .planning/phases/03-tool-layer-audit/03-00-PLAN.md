    -- Dependencies: none (standalone, runs antes de Phase 4 agent loop)

    CREATE TABLE IF NOT EXISTS jarvis_runs (
      id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
      session_id text NOT NULL,
      tenant_id uuid,
      status text NOT NULL DEFAULT 'pending'
        CHECK (status IN ('pending','running','completed','failed','needs_review')),
      started_at timestamptz NOT NULL DEFAULT now(),
      finished_at timestamptz,
      tokens_input integer NOT NULL DEFAULT 0,
      tokens_output integer NOT NULL DEFAULT 0,
      tokens_cache_read integer NOT NULL DEFAULT 0,
      tokens_cache_write integer NOT NULL DEFAULT 0,
      model text,

      error jsonb,
      created_at timestamptz NOT NULL DEFAULT now()
    );
    CREATE INDEX IF NOT EXISTS jarvis_runs_session_id_idx ON jarvis_runs (session_id);
    CREATE INDEX IF NOT EXISTS jarvis_runs_status_started_at_idx ON jarvis_runs (status, started_at DESC);

    CREATE TABLE IF NOT EXISTS jarvis_tool_calls (
      id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
      run_id uuid NOT NULL REFERENCES jarvis_runs(id) ON DELETE CASCADE,
      session_id text NOT NULL,
      tool_name text NOT NULL,
      input jsonb NOT NULL,
      output jsonb,
      duration_ms integer,
      idempotent_hit boolean NOT NULL DEFAULT false,
      error jsonb,
      langfuse_span_id text,
      created_at timestamptz NOT NULL DEFAULT now()
    );
    CREATE INDEX IF NOT EXISTS jarvis_tool_calls_run_id_idx ON jarvis_tool_calls (run_id);
    CREATE INDEX IF NOT EXISTS jarvis_tool_calls_session_id_created_at_idx ON jarvis_tool_calls (session_id, created_at DESC);
    CREATE INDEX IF NOT EXISTS jarvis_tool_calls_tool_name_idx ON jarvis_tool_calls (tool_name);

    CREATE TABLE IF NOT EXISTS idempotency_keys (
      session_id text NOT NULL,
      tool text NOT NULL,
      args_hash text NOT NULL,
      result jsonb NOT NULL,
      external_id text,
      created_at timestamptz NOT NULL DEFAULT now(),
      PRIMARY KEY (session_id, tool, args_hash)
    );
    CREATE INDEX IF NOT EXISTS idempotency_keys_created_at_idx ON idempotency_keys (created_at DESC);
