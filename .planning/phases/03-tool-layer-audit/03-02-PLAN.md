    ```typescript
    import { z } from 'zod';
    import { withIdempotency } from './idempotency';
    import { recordToolCall } from './audit';
    import { HttpError, type ToolContext } from './types';

    export type ToolHandler<I, O> = (ctx: ToolContext, args: I) => Promise<O>;

    export type ToolDefinition<I = unknown, O = unknown> = {
      name: string;
      description: string;
      inputSchema: z.ZodType<I>;
      outputSchema: z.ZodType<O>;
      handler: ToolHandler<I, O>;
      invoke: (ctx: ToolContext, rawArgs: unknown) => Promise<ToolInvokeResult<O>>;
    };

    export type ToolInvokeResult<O> =
      | { ok: true; data: O; idempotentHit: boolean; durationMs: number }
      | { ok: false; error: { code: string; message: string; retriable: boolean }; durationMs: number };

    export function wrapTool<I, O>(spec: {
      name: string;
      description: string;
      inputSchema: z.ZodType<I>;
      outputSchema: z.ZodType<O>;
      handler: ToolHandler<I, O>;
    }): ToolDefinition<I, O> {
      const invoke = async (ctx: ToolContext, rawArgs: unknown): Promise<ToolInvokeResult<O>> => {
        const start = Date.now();
        let idempotentHit = false;

        // 1) Validate input
        const parsed = spec.inputSchema.safeParse(rawArgs);
        if (!parsed.success) {
          const err = { code: 'invalid_input', message: parsed.error.message, retriable: false };
          await recordToolCall({
            runId: ctx.runId, sessionId: ctx.sessionId, toolName: spec.name,
            input: rawArgs, durationMs: Date.now() - start, idempotentHit: false, error: err,
          });
          return { ok: false, error: err, durationMs: Date.now() - start };
        }

        try {
          // 2) Run handler under idempotency (also validates output)
          const result = await withIdempotency(
            { sessionId: ctx.sessionId, tool: spec.name },
            parsed.data,
            async () => {
              const raw = await spec.handler(ctx, parsed.data);
              const outParsed = spec.outputSchema.safeParse(raw);
              if (!outParsed.success) {
                throw new HttpError('http_4xx', null, false, `invalid_output: ${outParsed.error.message}`);
              }
              return outParsed.data;
            },
          );
          idempotentHit = result.idempotentHit;

          await recordToolCall({
            runId: ctx.runId, sessionId: ctx.sessionId, toolName: spec.name,
            input: parsed.data, output: result.data,
            durationMs: Date.now() - start, idempotentHit,
          });
          return { ok: true, data: result.data, idempotentHit, durationMs: Date.now() - start };
        } catch (e) {
          const httpErr = e instanceof HttpError;
          const err = {
            code: httpErr ? (e as HttpError).code : 'handler_error',
            message: (e as Error).message,
            retriable: httpErr ? (e as HttpError).retriable : false,
          };
          await recordToolCall({
            runId: ctx.runId, sessionId: ctx.sessionId, toolName: spec.name,
            input: parsed.data, durationMs: Date.now() - start, idempotentHit, error: err,
          });
          return { ok: false, error: err, durationMs: Date.now() - start };
        }
      };

      return { ...spec, invoke };
    }
    ```

      outputSchema: CreateTenantOutput,
      handler: async (ctx, args) => {
        const res = await callExternal<O>({
          url: `${PIPEELO_API_BASE}/tenants`,
          method: 'POST',
          headers: {

            'Idempotency-Key': `${ctx.sessionId}:create_tenant`,
          },
          body: args,
          timeoutMs: 20000,
        });
        return res.data;
      },
    });
