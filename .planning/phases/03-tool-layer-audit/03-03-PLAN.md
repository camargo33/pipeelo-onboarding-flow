    ```typescript
    import { Langfuse } from 'langfuse';

    let cached: Langfuse | null | undefined;

    export function getLangfuseClient(): Langfuse | null {
      if (cached !== undefined) return cached;
      const publicKey = process.env.LANGFUSE_PUBLIC_KEY;
      const secretKey = process.env.LANGFUSE_SECRET_KEY;
      if (!publicKey || !secretKey) {
        console.warn('[langfuse] keys missing — operating in no-op mode');
        cached = null;
        return null;
      }
      cached = new Langfuse({
        publicKey, secretKey,
        baseUrl: process.env.LANGFUSE_HOST ?? 'https://cloud.langfuse.com',
      });
      return cached;
    }

    export function __resetLangfuseCache() { cached = undefined; }

    export type JarvisTrace = {
      traceId: string | null;
      span: (input: { name: string; input?: unknown; metadata?: Record<string, unknown> }) => JarvisSpan;
      end: (output?: unknown) => void;
    };

    export type JarvisSpan = {
      spanId: string | null;
      end: (output?: unknown, level?: 'DEFAULT' | 'ERROR') => void;
    };

    export function createTrace(input: {
      name: string;
      sessionId: string;
      tenantId?: string;
      metadata?: Record<string, unknown>;
    }): JarvisTrace {
      const lf = getLangfuseClient();
      if (!lf) {
        return {
          traceId: null,
          span: () => ({ spanId: null, end: () => undefined }),
          end: () => undefined,
        };
      }
      const trace = lf.trace({
        name: input.name,
        sessionId: input.sessionId,
        tags: input.tenantId ? [`tenant:${input.tenantId}`] : [],
        metadata: { ...input.metadata, tenant_id: input.tenantId },
      });
      return {
        traceId: trace.id,
        span: ({ name, input: spanInput, metadata }) => {
          const s = trace.span({ name, input: spanInput, metadata });
          return {
            spanId: s.id,
            end: (output, level) => {
              try { s.end({ output, level: level ?? 'DEFAULT' }); }
              catch (e) { console.error('[langfuse] span.end failed:', (e as Error).message); }
            },
          };
        },
        end: (output) => { try { trace.update({ output }); } catch {} },
      };
    }

    export async function withSpan<T>(
      trace: JarvisTrace | null | undefined,
      name: string,
      input: unknown,
      fn: () => Promise<T>,
      metadata?: Record<string, unknown>,
    ): Promise<{ result: T; spanId: string | null }> {
      const span = trace?.span({ name, input, metadata }) ?? { spanId: null, end: () => undefined };
      try {
        const result = await fn();
        span.end({ ok: true }, 'DEFAULT');
        return { result, spanId: span.spanId };
      } catch (e) {
        span.end({ ok: false, error: (e as Error).message }, 'ERROR');
        throw e;
      }
    }

    export async function flushLangfuse(): Promise<void> {
      const lf = getLangfuseClient();
      if (!lf) return;
      try { await lf.flushAsync(); } catch (e) { console.error('[langfuse] flush failed:', (e as Error).message); }
    }
    ```

    ```typescript
    import { getJarvisSupabase } from '@/api/jarvis/_runtime/tools/_shared/supabase';

    export async function GET(_: NextRequest, { params }: { params: Promise<{ id: string }> }) {
      const { id } = await params;
      const supabase = getJarvisSupabase();
      const [{ data: run, error: runErr }, { data: calls, error: callsErr }] = await Promise.all([
        supabase.from('jarvis_runs').select('*').eq('id', id).single(),
        supabase.from('jarvis_tool_calls').select('*').eq('run_id', id).order('created_at', { ascending: true }),
      ]);
      if (runErr) return NextResponse.json({ error: runErr.message }, { status: 404 });
      if (callsErr) return NextResponse.json({ error: callsErr.message }, { status: 500 });
      return NextResponse.json({ run, tool_calls: calls });
    }
    ```
